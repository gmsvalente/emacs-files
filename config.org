* Export to README.md
#+begin_src emacs-lisp
  (defun config.org->README.md ()
    (find-file config-file)
    (switch-to-buffer "config.org")
    (org-md-export-as-markdown)
    (switch-to-buffer "*Org MD Export*")
    (write-file "~/.emacs.d/README.md"))
#+end_src
